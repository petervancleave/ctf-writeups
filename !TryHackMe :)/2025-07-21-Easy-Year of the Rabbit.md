            ptr += 1
        elif cmd == '<':
            ptr -= 1
        elif cmd == '+':
            tape[ptr] = (tape[ptr] + 1) % 256
        elif cmd == '-':
            tape[ptr] = (tape[ptr] - 1) % 256
        elif cmd == '.':

        elif cmd == ']':

                i = loop_stack[-1]
                continue
            else:
                loop_stack.pop()
        i += 1

    return output
